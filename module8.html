📦 Installing the requests Library

Unlike socket, os, and sys, the requests library is not built-in — you install it once and it handles all HTTP complexity for you.

terminal – install requests 
# Install using pip
pip3 install requests

# Then import in your Python script
import requests
🏆 Why requests over raw sockets for HTTP? Raw sockets require you to manually build HTTP headers, handle redirects, decode responses, and manage cookies. The requests library handles all of that automatically in one line.
🌐 Making HTTP Requests
python – basic http requests 
import requests

# GET request — fetch a page
r = requests.get("http://httpbin.org/get")

# Status code
print(r.status_code)      # 200

# Response headers
print(r.headers)          # dict of headers
print(r.headers["Server"])  # e.g. "nginx"

# Response body
print(r.text)             # full HTML/text
print(r.json())           # parse JSON automatically

# With timeout — always use this!
r = requests.get("http://target.com", timeout=3)

# Custom headers — disguise your user agent
headers = {"User-Agent": "Mozilla/5.0 (compatible)"}
r = requests.get("http://target.com", headers=headers)

🔍 Header Analysis — Fingerprint a Server

HTTP response headers reveal critical information about a server — what software it runs, security configurations, and misconfigurations.

python – header analyser 
import requests

def analyse_headers(url):
    try:
        r = requests.get(url, timeout=5, verify=False)
        h = r.headers

        print(f"\n🔍 Header Analysis: {url}")
        print("=" * 45)
        print(f"  Status       : {r.status_code}")
        print(f"  Server       : {h.get('Server', 'Hidden')}")
        print(f"  Powered-By   : {h.get('X-Powered-By', 'Hidden')}")
        print(f"  Content-Type : {h.get('Content-Type', 'N/A')}")

        # Security header checks
        print("\n  Security Headers:")
        sec = [
            ("X-Frame-Options",   "Clickjacking protection"),
            ("X-XSS-Protection",  "XSS filter"),
            ("Strict-Transport-Security", "HSTS"),
            ("Content-Security-Policy",   "CSP"),
        ]
        for hdr, desc in sec:
            present = hdr in h
            icon = "✅" if present else "⚠️ "
            print(f"  {icon} {desc:28} {'SET' if present else 'MISSING'}")
        print("=" * 45)

    except requests.exceptions.RequestException as e:
        print(f"❌ Error: {e}")

analyse_headers("http://httpbin.org")

🗂️ Mini Project — Directory Brute Forcer

This tool tries common directory names on a web server — a core technique in web application pentesting:

python – dir_bruteforce.py 
import requests

BASE_URL = "http://testphp.vulnweb.com"  # legal test site
WORDLIST = [
    "admin", "login", "dashboard",
    "backup", "config", "uploads",
    "api", "test", "hidden", "secret"
]

found = []

print(f"[*] Target  : {BASE_URL}")
print(f"[*] Testing : {len(WORDLIST)} directories\n")

for path in WORDLIST:
    url = f"{BASE_URL}/{path}"
    try:
        r = requests.get(url, timeout=3, allow_redirects=False)
        if r.status_code == 200:
            print(f"  ✅ FOUND  [{r.status_code}] → /{path}")
            found.append(url)
        elif r.status_code == 403:
            print(f"  🔒 FORBID [{r.status_code}] → /{path}")
        else:
            print(f"  ❌       [{r.status_code}] → /{path}")
    except:
        print(f"  ⚡ TIMEOUT → /{path}")

print(f"\n[+] {len(found)} accessible paths found:")
for u in found:
    print(f"    → {u}")
